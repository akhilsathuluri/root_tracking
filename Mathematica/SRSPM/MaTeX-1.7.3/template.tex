\newcommand{\MaTeX}[1]{
	\begin{matex}
		\setbox\MaTeXbox\hbox{%
			`strut`%
			\(%
				`display`%
				#1%
			\)}%
		\typeout{MATEXDEPTH:\the\dp\MaTeXbox}%
		\typeout{MATEXHEIGHT:\the\ht\MaTeXbox}%
		\typeout{MATEXWIDTH:\the\wd\MaTeXbox}%
		\unhbox\MaTeXbox%
	\end{matex}
}
